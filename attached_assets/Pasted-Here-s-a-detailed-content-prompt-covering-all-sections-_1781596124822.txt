### 🔧 Functional Implementation (Next.js + Nodemailer / EmailJS)

**Option A — EmailJS (zero backend, instant setup):**
```js
// Install: npm install @emailjs/browser

import emailjs from '@emailjs/browser';

const sendEmail = async (formData) => {
  await emailjs.send(
    'YOUR_SERVICE_ID',       // Gmail service
    'YOUR_TEMPLATE_ID',      // Template with fields below
    {
      from_name:    formData.name,
      company:      formData.company,
      phone:        formData.phone,
      email:        formData.email,
      city:         formData.city,
      service:      formData.service,
      staff_count:  formData.staffCount,
      message:      formData.message,
      to_email:     'assistanceprotectionservices@gmail.com'
    },
    'YOUR_PUBLIC_KEY'
  );
};

// EmailJS Template text:
/*
New Enquiry from APS Website

Name: {{from_name}}
Company: {{company}}
Phone: {{phone}}
Email: {{email}}
City: {{city}}
Service: {{service}}
Staff Required: {{staff_count}}
Message: {{message}}
*/
```

**Option B — Next.js API Route + Nodemailer:**
```js
// /app/api/contact/route.ts
import nodemailer from 'nodemailer';

export async function POST(req: Request) {
  const body = await req.json();

  const transporter = nodemailer.createTransport({
    service: 'gmail',
    auth: {
      user: process.env.SMTP_USER,   // your gmail
      pass: process.env.SMTP_PASS,   // app password
    },
  });

  await transporter.sendMail({
    from: body.email,
    to: 'assistanceprotectionservices@gmail.com',
    subject: `New Enquiry: ${body.service} — ${body.name}`,
    html: `
      <h2>New Website Enquiry</h2>
      <p><b>Name:</b> ${body.name}</p>
      <p><b>Company:</b> ${body.company}</p>
      <p><b>Phone:</b> ${body.phone}</p>
      <p><b>Email:</b> ${body.email}</p>
      <p><b>City:</b> ${body.city}</p>
      <p><b>Service:</b> ${body.service}</p>
      <p><b>Staff Required:</b> ${body.staffCount}</p>
      <p><b>Message:</b> ${body.message}</p>
    `,
  });

  return Response.json({ success: true });
}
```
